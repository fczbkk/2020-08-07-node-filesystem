- Vytvorte logger, ktorý bude pripájať obsah k nejakému súboru.
- Extra body dostanete za limit obsahu súboru na 10 riadkov.

JS:

myLogger('aaa')
myLogger('bbb')

my-log.txt:

aaa
bbb



JS:

myLogger('aaa')
myLogger('bbb')
myLogger('ccc')
myLogger('ddd')
myLogger('eee')
myLogger('fff')
myLogger('ggg')
myLogger('hhh')
myLogger('iii')
myLogger('jjj')
myLogger('kkk')
myLogger('lll')

my-log.txt:

ccc
ddd
eee
fff
ggg
hhh
iii
jjj
kkk
lll
